% work-around to have small caps also here in the headline
% https://tex.stackexchange.com/questions/188126/wrong-header-in-bibliography-classicthesis
% Thanks to Enrico Gregorio
\defbibheading{bibintoc}[References]{%
  \phantomsection
  \manualmark
  \markboth{\spacedlowsmallcaps{References}}{\spacedlowsmallcaps{References}}%
  \addtocontents{toc}{\protect\vspace{\beforebibskip}}%
  \addcontentsline{toc}{part}{References}%
  % \addcontentsline{toc}{chapter}{\tocEntry{References}}%
  \chapter*{References}%
}
% Force bibliography entries to wrap if they exceed the margin
\setlength{\bibhang}{1.5em} % hanging indent
\sloppy % allow line breaks to avoid overfull hboxes
\setlength{\baselineskip}{1.5\baselineskip} % set line spacing to 1.5
\printbibliography[heading=bibintoc]
